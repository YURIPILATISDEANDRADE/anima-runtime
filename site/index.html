import core;

// For toString
using core.Convert;
// For writeLine
using core.IO;

namespace Hello {
    macro greet {
        { $name : string } => { ( "Hello, " + ( $name + "!" ) ) }
        { $name : expr } => { ( "Hello, " + ( toString( $name ) + "!" ) ) }
    }
}

main {
    writeLine(Hello.greet "someone");
}
